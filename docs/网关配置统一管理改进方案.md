# 网关配置统一管理改进方案

> 版本: v1.0  
> 创建日期: 2026-03-24  
> 状态: 待实施

---

## 一、背景与目标

### 1.1 现状问题

当前网关服务(gateway-service)的配置存在以下问题：

1. **配置分散**: 配置分布在 `application.yml`、Java常量类、`@Value`注解多处
2. **无法动态调整**: 修改配置需要重启网关服务
3. **配置不一致风险**: 网关与监测服务的配置可能不同步
4. **运维困难**: 无法通过统一界面管理整个系统的配置

### 1.2 改进目标

1. **统一配置管理**: 所有配置由监测服务统一管理，存储在数据库中
2. **动态生效**: 配置变更后实时推送到网关，无需重启
3. **可视化配置**: 通过前端界面统一管理所有配置
4. **高可用设计**: 网关具备本地缓存，监测服务不可用时仍可正常运行

## 三、目标架构

### 3.1 架构设计

```
┌─────────────────┐     1.启动拉取      ┌─────────────────────────────────┐
│   网关服务       │ ─────────────────→ │         监测服务                 │
│                 │                     │                                 │
│  ┌───────────┐  │     2.推送更新      │  ┌───────────────────────────┐  │
│  │ Gateway   │  │ ←─────────────────  │  │     系统配置管理           │  │
│  │ Config    │  │                     │  │  - SysConfigService       │  │
│  │ Cache     │  │     3.心跳同步      │  │  - SysConfigCache         │  │
│  │           │  │ ←─────────────────  │  │  - GatewayConfigService   │  │
│  └───────────┘  │                     │  └───────────────────────────┘  │
│                 │                     │                                 │
│  ┌───────────┐  │                     │  ┌───────────────────────────┐  │
│  │ 过滤器    │  │                     │  │     数据库                 │  │
│  │ 使用配置  │  │                     │  │  sys_config 表            │  │
│  └───────────┘  │                     │  │  存储所有配置              │  │
└─────────────────┘                     └─────────────────────────────────┘
```

### 3.2 配置同步机制

| 时机 | 方式 | 说明 |
|------|------|------|
| 网关启动 | 主动拉取 | 从监测服务获取全部配置 |
| 配置变更 | 被动接收 | 监测服务推送配置更新 |
| 定时同步 | 心跳检查 | 定期同步确保一致性 |
| 配置失效 | 本地默认 | 网络异常时使用本地缓存 |

#### 2.2 新增服务实现

**文件路径**: `monitor-service/src/main/java/com/network/monitor/service/impl/GatewayConfigServiceImpl.java`

```java
package com.network.monitor.service.impl;

import com.network.monitor.cache.SysConfigCache;
import com.network.monitor.client.GatewayApiClient;
import com.network.monitor.service.GatewayConfigService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.Arrays;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

@Slf4j
@Service
public class GatewayConfigServiceImpl implements GatewayConfigService {

    private static final String GATEWAY_CONFIG_PREFIX = "gateway.";

    @Autowired
    private SysConfigCache sysConfigCache;

    @Autowired
    private GatewayApiClient gatewayApiClient;

    @Override
    public Map<String, Object> getAllGatewayConfigs() {
        Map<String, Object> configs = new HashMap<>();
        
        // 定义所有网关配置键
        List<String> gatewayConfigKeys = Arrays.asList(
            "gateway.defense.blacklist.enabled",
            "gateway.defense.rate-limit.enabled",
            "gateway.defense.malicious-request.enabled",
            "gateway.defense.rate-limit.default-threshold",
            "gateway.defense.rate-limit.window-size",
            "gateway.defense.blacklist.default-expire-seconds",
            "gateway.defense.malicious.user-agents",
            "gateway.defense.malicious.uri-patterns",
            "gateway.cache.traffic-expire-ms",
            "gateway.cache.blacklist-expire-ms",
            "gateway.cache.cleanup-interval-ms",
            "gateway.attack-state.cooldown-duration-ms",
            "gateway.attack-state.state-expire-ms",
            "gateway.request.max-body-size",
            "gateway.request.abnormal-response-threshold-ms"
        );

        for (String key : gatewayConfigKeys) {
            String value = sysConfigCache.getConfigValue(key);
            if (value != null) {
                configs.put(key, value);
            }
        }

        return configs;
    }

    @Override
    public String getConfigValue(String configKey, String defaultValue) {
        return sysConfigCache.getConfigValue(configKey, defaultValue);
    }

    @Override
    public int getIntConfigValue(String configKey, int defaultValue) {
        return sysConfigCache.getIntValue(configKey, defaultValue);
    }

    @Override
    public boolean getBooleanConfigValue(String configKey, boolean defaultValue) {
        return sysConfigCache.getBooleanValue(configKey, defaultValue);
    }

    @Override
    public List<String> getListConfigValue(String configKey, String defaultValue) {
        String value = sysConfigCache.getConfigValue(configKey, defaultValue);
        if (value == null || value.isEmpty()) {
            return Arrays.asList();
        }
        return Arrays.asList(value.split(","));
    }

    @Override
    public void updateConfig(String configKey, String configValue) {
        sysConfigCache.updateConfig(configKey, configValue);
        pushConfigToGateway(configKey, configValue);
    }

    @Override
    public boolean pushConfigToGateway(String configKey, String configValue) {
        try {
            return gatewayApiClient.pushConfigToGateway(configKey, configValue);
        } catch (Exception e) {
            log.error("推送配置到网关失败: key={}, value={}", configKey, configValue, e);
            return false;
        }
    }

    @Override
    public boolean pushAllConfigsToGateway() {
        try {
            Map<String, Object> configs = getAllGatewayConfigs();
            return gatewayApiClient.pushConfigsToGateway(configs);
        } catch (Exception e) {
            log.error("推送所有配置到网关失败", e);
            return false;
        }
    }

    @Override
    public void refreshGatewayConfigCache() {
        sysConfigCache.refresh();
        pushAllConfigsToGateway();
    }
}
```

package com.network.monitor.controller.inner;

import com.network.monitor.common.ApiResponse;
import com.network.monitor.dto.GatewayConfigDTO;
import com.network.monitor.service.GatewayConfigService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

import java.util.Map;

/**
 * 网关配置同步接口
 * 供网关服务调用
 */
@Slf4j
@RestController
@RequestMapping("/api/inner/gateway/config")
public class GatewayConfigController {

    @Autowired
    private GatewayConfigService gatewayConfigService;

    /**
     * 网关拉取全部配置
     * 网关启动时调用此接口获取所有配置
     */
    @GetMapping("/sync")
    public ApiResponse<Map<String, Object>> syncConfigs() {
        try {
            Map<String, Object> configs = gatewayConfigService.getAllGatewayConfigs();
            log.info("网关拉取配置成功，共{}项", configs.size());
            return ApiResponse.success(configs);
        } catch (Exception e) {
            log.error("网关拉取配置失败", e);
            return ApiResponse.error("获取配置失败");
        }
    }

    /**
     * 推送配置到网关
     * 管理员手动触发
     */
    @PostMapping("/push")
    public ApiResponse<Void> pushConfigToGateway(@RequestBody GatewayConfigDTO dto) {
        try {
            boolean success;
            if (dto.getConfigs() != null && !dto.getConfigs().isEmpty()) {
                success = gatewayConfigService.pushAllConfigsToGateway();
            } else {
                success = gatewayConfigService.pushConfigToGateway(dto.getConfigKey(), dto.getConfigValue());
            }
            
            if (success) {
                return ApiResponse.success();
            } else {
                return ApiResponse.error("推送配置失败");
            }
        } catch (Exception e) {
            log.error("推送配置到网关失败", e);
            return ApiResponse.error("推送配置失败: " + e.getMessage());
        }
    }

    /**
     * 刷新网关配置缓存
     */
    @PostMapping("/refresh")
    public ApiResponse<Void> refreshConfig() {
        try {
            gatewayConfigService.refreshGatewayConfigCache();
            return ApiResponse.success();
        } catch (Exception e) {
            log.error("刷新网关配置缓存失败", e);
            return ApiResponse.error("刷新配置失败");
        }
    }
}

public boolean pushConfigToGateway(String configKey, String configValue) {
    try {
        String url = GATEWAY_BASE_URL + "/api/gateway/config/push";
        
        HttpHeaders headers = new HttpHeaders();
        headers.setContentType(MediaType.APPLICATION_JSON);
        headers.set("X-Auth-Token", gatewayAuthToken);
        headers.set("X-Source-IP", monitorIp);
        
        Map<String, String> config = new HashMap<>();
        config.put("configKey", configKey);
        config.put("configValue", configValue);
        config.put("timestamp", String.valueOf(System.currentTimeMillis()));
        
        HttpEntity<Map<String, String>> request = new HttpEntity<>(config, headers);
        
        ResponseEntity<String> response = restTemplate.postForEntity(url, request, String.class);
        
        if (response.getStatusCode().is2xxSuccessful()) {
            log.info("推送配置到网关成功: key={}", configKey);
            return true;
        } else {
            log.error("推送配置到网关失败: key={}, statusCode={}", configKey, response.getStatusCode());
            return false;
        }
    } catch (Exception e) {
        log.error("推送配置到网关异常: key={}", configKey, e);
        return false;
    }
}

    } catch (Exception e) {

import com.network.gateway.client.MonitorServiceConfigClient;

    // 默认配置值

import org.springframework.stereotype.Component;

 * 监测服务配置拉取客户端
 */
@Slf4j
@Component
public class MonitorServiceConfigClient {

    @Autowired
    @Qualifier("restTemplate")
    private RestTemplate restTemplate;

    @Value("${cross-service.security.secret-key}")
    private String secretKey;

    @Autowired
    private ObjectMapper objectMapper;

    /**
     * 从监测服务拉取配置
     */
    @SuppressWarnings("unchecked")
    public Map<String, Object> pullConfigs() {
        try {
            String url = GatewayHttpConstant.MonitorService.BASE_URL + "/api/inner/gateway/config/sync";
            
            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.APPLICATION_JSON);
            
            String requestId = java.util.UUID.randomUUID().toString().replace("-", "").substring(0, 16);
            Map<String, String> securityHeaders = CrossServiceSecurityUtil.generateSecurityHeaders(
                    requestId, secretKey);
            headers.set("X-Timestamp", securityHeaders.get("X-Timestamp"));
            headers.set("X-Request-ID", securityHeaders.get("X-Request-ID"));
            headers.set("X-Signature", securityHeaders.get("X-Signature"));
            
            HttpEntity<Void> request = new HttpEntity<>(headers);
            
            ResponseEntity<String> response = restTemplate.exchange(
                    url, HttpMethod.GET, request, String.class);
            
            if (response.getStatusCode().is2xxSuccessful() && response.getBody() != null) {
                Map<String, Object> responseMap = objectMapper.readValue(response.getBody(), Map.class);
                if (responseMap.containsKey("data")) {
                    return (Map<String, Object>) responseMap.get("data");
                }
            }
            
            log.error("从监测服务拉取配置失败: statusCode={}", response.getStatusCode());
            return null;
            
        } catch (Exception e) {
            log.error("从监测服务拉取配置异常: {}", e.getMessage());
            return null;
        }
    }
}

package com.network.gateway.controller;

import com.network.gateway.cache.GatewayConfigCache;
import com.network.gateway.exception.GatewayBizException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import org.springframework.http.ResponseEntity;
import org.springframework.util.StringUtils;
import org.springframework.web.bind.annotation.*;

import java.util.HashMap;
import java.util.Map;
import java.util.Objects;

/**
 * 网关配置同步接口
 * 接收监测服务推送的配置更新
 */
@RestController
@RequestMapping("/api/gateway/config")
public class GatewayConfigSyncController {

    private static final Logger logger = LoggerFactory.getLogger(GatewayConfigSyncController.class);

    @Value("${gateway.defense.auth.token:NetMonitor2026SecureToken}")
    private String configuredAuthToken;

    @Value("${gateway.defense.auth.allowed-ips:127.0.0.1,localhost}")
    private String allowedMonitorServiceIps;

    @Autowired
    private GatewayConfigCache configCache;

    /**
     * 接收单个配置更新
     */
    @PostMapping("/push")
    public ResponseEntity<Map<String, Object>> receiveConfigPush(
            @RequestBody Map<String, String> payload,
            @RequestHeader(value = "X-Auth-Token", required = false) String authToken,
            @RequestHeader(value = "X-Source-IP", required = false) String sourceIp) {
        
        Map<String, Object> response = new HashMap<>();
        
        try {
            // 鉴权验证
            if (!validateAuthToken(authToken)) {
                response.put("success", false);
                response.put("message", "授权Token无效");
                return ResponseEntity.status(401).body(response);
            }

            if (!validateSourceIp(sourceIp)) {
                response.put("success", false);
                response.put("message", "请求来源IP不被允许");
                return ResponseEntity.status(403).body(response);
            }

            String configKey = payload.get("configKey");
            String configValue = payload.get("configValue");

            if (!StringUtils.hasText(configKey)) {
                response.put("success", false);
                response.put("message", "配置键不能为空");
                return ResponseEntity.badRequest().body(response);
            }

            configCache.updateConfig(configKey, configValue);

            response.put("success", true);
            response.put("message", "配置更新成功");
            response.put("configKey", configKey);
            logger.info("接收配置推送成功: key={}, value={}", configKey, configValue);
            
            return ResponseEntity.ok(response);

        } catch (Exception e) {
            logger.error("处理配置推送异常", e);
            response.put("success", false);
            response.put("message", "处理配置推送异常: " + e.getMessage());
            return ResponseEntity.status(500).body(response);
        }
    }

    /**
     * 批量同步配置
     */
    @PostMapping("/sync")
    public ResponseEntity<Map<String, Object>> syncConfigs(
            @RequestBody Map<String, Object> payload,
            @RequestHeader(value = "X-Auth-Token", required = false) String authToken,
            @RequestHeader(value = "X-Source-IP", required = false) String sourceIp) {
        
        Map<String, Object> response = new HashMap<>();
        
        try {
            // 鉴权验证
            if (!validateAuthToken(authToken)) {
                response.put("success", false);
                response.put("message", "授权Token无效");
                return ResponseEntity.status(401).body(response);
            }

            if (!validateSourceIp(sourceIp)) {
                response.put("success", false);
                response.put("message", "请求来源IP不被允许");
                return ResponseEntity.status(403).body(response);
            }

            @SuppressWarnings("unchecked")
            Map<String, String> configs = (Map<String, String>) payload.get("configs");

            if (configs == null || configs.isEmpty()) {
                response.put("success", false);
                response.put("message", "配置数据为空");
                return ResponseEntity.badRequest().body(response);
            }

            configCache.updateConfigs(configs);

            response.put("success", true);
            response.put("message", "配置同步成功");
            response.put("count", configs.size());
            logger.info("批量同步配置成功，共{}项", configs.size());
            
            return ResponseEntity.ok(response);

        } catch (Exception e) {
            logger.error("批量同步配置异常", e);
            response.put("success", false);
            response.put("message", "批量同步配置异常: " + e.getMessage());
            return ResponseEntity.status(500).body(response);
        }
    }

    /**
     * 获取当前配置
     */
    @GetMapping("/current")
    public ResponseEntity<Map<String, Object>> getCurrentConfigs() {
        Map<String, Object> response = new HashMap<>();
        response.put("success", true);
        response.put("configs", configCache.getAllConfigs());
        response.put("count", configCache.getConfigCount());
        return ResponseEntity.ok(response);
    }

    /**
     * 手动刷新配置
     */
    @PostMapping("/refresh")
    public ResponseEntity<Map<String, Object>> refreshConfigs(
            @RequestHeader(value = "X-Auth-Token", required = false) String authToken) {
        
        Map<String, Object> response = new HashMap<>();
        
        if (!validateAuthToken(authToken)) {
            response.put("success", false);
            response.put("message", "授权Token无效");
            return ResponseEntity.status(401).body(response);
        }

        try {
            configCache.pullFromMonitorService();
            response.put("success", true);
            response.put("message", "配置刷新成功");
            response.put("count", configCache.getConfigCount());
            return ResponseEntity.ok(response);
        } catch (Exception e) {
            logger.error("刷新配置异常", e);
            response.put("success", false);
            response.put("message", "刷新配置异常: " + e.getMessage());
            return ResponseEntity.status(500).body(response);
        }
    }

    private boolean validateAuthToken(String authToken) {
        if (!StringUtils.hasText(authToken)) {
            return false;
        }
        return Objects.equals(configuredAuthToken, authToken);
    }

    private boolean validateSourceIp(String sourceIp) {
        if (!StringUtils.hasText(sourceIp)) {
            return true;
        }
        String[] allowedIps = allowedMonitorServiceIps.split(",");
        for (String allowedIp : allowedIps) {
            if (allowedIp.trim().equals(sourceIp)) {
                return true;
            }
        }
        return false;
    }
